 10  #include <exception>
 11  #include <iostream>
 12  #include <string>
 13  
 14  template <class T, int ARRAYSIZE = 1024> class List {
 15      private:
 16          T** data;
 17          int last;
 18  
 19          void copyAll(const List& l) {
 20              deleteAll();
 21  
 22              for(last = -1; last < l.last; data[++last] = new T(*l.data[last])) if (data[last] == nullptr) throw ListException("Something went wrong in List constructor, memory not available");
 23              }
 24  
 25          bool isValidPos(const int& p) {
 26              return p >= 0 or p <= last;
 27              }
 28  
 29          void sortData(const int& l, const int& r) {
 30              if(l >= r)
 31                  return;
 32  
 33              if (l + 1 == r) {
 34                  if(*data[l] > *data[r]) std::swap(data[l], data[r]);
 35  
 36                  return;
 37                  }
 38  
 39              int i = l, j = r;
 40  
 41              while(i < j) {
 42                  while(i < j and *data[i] <= *data[r]) i++;
 43                  while(i < j and *data[j] >= *data[r]) j--;
 44  
 45                  if(i != j) std::swap(data[i], data[j]);
 46                  }
 47  
 48              if(i != r) std::swap(data[i], data[r]);
 49  
 50              if(i > l) sortData(l, i - 1);
 51  
 52              if(i < j) sortData(i + 1, j);
 53              }
 54  
 55      public:
 56          class ListException : public std::exception {
 57              private:
 58                  std::string msg;
 59  
 60              public:
 61                  explicit ListException(const char* message) : msg(message) { }
 62                  explicit ListException(const std::string& message) : msg(message) { }
 63                  virtual ~ListException() throw () { }
 64                  virtual const char* what() const throw () { return msg.c_str(); }
 65              };
 66  
 67          List() {
 68              if((data = new T*[ARRAYSIZE]) == nullptr) throw ListException("Something went wrong in List constructor, memory not available");
 69  
 70  
 71              for(last = ARRAYSIZE; last >= 0; data[--last] = nullptr);
 72              }
 73  
 74          List(const List& l) : List() {
 75              copyAll(l);
 76              }
 77  
 78          ~List() {
 79              deleteAll();
 80  
 81              delete[] *data;
 82              }
 83  
 84          bool isEmpty() {
 85              return last == -1;
 86              }
 87  
 88          bool isFull() {
 89              return last == ARRAYSIZE - 1;
 90              }
 91  
 92          void insertData(int p, const T& e) {
 93              if(isFull()) throw ListException("Can not insert data in a full List");
 94  
 95              if(p != -1 and !isValidPos(p)) throw ListException("There is an invalid position, trying to insert data into List");
 96  
 97              for(int i = last++; i > p; data[i + 1] = data[i]);
 98  
 99              if((data[p + 1] = new T(e)) == nullptr) throw ListException("Something went wrong inserting new data in List");
100              }
101  
102          void insertData(const T& e) {
103              insertData(getLastPos(), e);
104              }
105  
106          void deleteData(int p) {
107              if(!isValidPos(p)) throw ListException("There is an invalid position, trying to delete data from List");
108  
109              for(int i = p; i < last; data[i] = data[++i]);
110  
111              delete data[last--];
112              }
113  
114          int getFirstPos() {
115              return last == -1 ? -1 : 0;
116              }
117  
118          int getLastPos() {
119              return last;
120              }
121  
122          int getPrevPos(const int& p) {
123              return p == 0 or !isValidPos(p) ? -1 : p - 1;
124              }
125  
126          int getNextPos(const int& p) {
127              return p == last or !isValidPos(p) ? -1 : p + 1;
128              }
129  
130          int findData(const T& e) {
131              for(int i = 0; i <= last; i++) if(data[i] == e) return i;
132  
133              return -1;
134              }
135  
136          T retrieveData(const int p) {
137              if(!isValidPos(p)) throw ListException("Invalid position, trying to retrieve data from List");
138  
139              return *data[p];
140              }
141  
142          void sortData() {
143              sortData(0, last);
144              }
145  
146          void printData() {
147              for(int i = 0; i <= last; i++) std::cout << *data[i] << std::endl;
148              }
149  
150          void deleteAll() {
151              for( ; last >= 0; delete data[last], data[last--] = nullptr);
152              }
153  
154          List& operator = (const List& l) {
155              deleteAll();
156  
157              copyAll(l);
158  
159              return *this;
160              }
161  
162          friend std::ostream& operator << (std::ostream& os, List& l) {
163              for(int i = 0; i <= l.last; i++) os << l.data[i] << std::endl;
164  
165              return os;
166              }

169              T myT;
170              int i = 0;
171              while (is >> myT)
172                  if((l.data[i++] = new T(myT)) == nullptr) throw ListException("There is an invalid position, trying to insert data into List in >> operator");
173  
174              return is;
175              }
